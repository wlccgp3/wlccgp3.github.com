p = figure(
    plot_width=400,  # 绘图宽度
    plot_height=400,  # 绘图高度
    title='标题',
    x_range=(-2, 5),  # x轴显示范围
)

p.xaxis.axis_label = "Temp"  # x轴标签
p.xaxis.axis_line_width = 3  # x轴线宽
p.xaxis.axis_line_color = "red"  # x轴颜色

p.axis.minor_tick_in = 5  # 所有坐标轴细刻度向内长度
p.axis.minor_tick_out = 10  # 所有坐标轴细刻度向外长度
p.axis.major_tick_in = 10  # 所有坐标轴主刻度向内长度
p.axis.major_tick_out = 20  # 所有坐标轴主刻度向外长度

p.yaxis.axis_line_dash = 'dashdot'  # y轴轴线样式
p.yaxis.axis_label = "Pressure"  # y轴标签
p.yaxis.major_label_text_color = "green"  # y轴主刻度标签颜色
p.yaxis.major_label_orientation = "vertical"  # y轴主刻度标签竖直显示

p.outline_line_width = 5  # 边框宽度
p.outline_line_alpha = 0.3  # 边框透明度
p.outline_line_color = "navy"  # 边框颜色

p.xgrid.grid_line_color = None  # 取消x轴网格
p.xaxis.ticker = [2, 3.5, 4]  # x轴主刻度范围

p.circle([1,2,3,4,5],
         [2,5,8,2,7],
         size=[10, 10, 10, 15, 20],  # 为每一个标记设定大小，长度和标记一样
         line_color="red",  # 标记边线颜色
         color="white",  # 一次性填充颜色，包括边线，优先级小于line_color和fill_color等设定颜色的参数
         legend='图例')

p.legend.location = 'bottom_right'  # 图例位置
p.legend.click_policy = 'hide'  # 图例交互，点击显示或隐藏  

show(p)

显示图片¶

The example below shows how to use the image_rgba method to display raw RGBA data.

Note: This example makes use of the NumPy library

In [11]:
import numpy as np

N = 10
img = np.empty((N,N), dtype=np.uint32)
view = img.view(dtype=np.uint8).reshape((N, N, 4))
for i in range(N):
    for j in range(N):
        view[i, j, 0] = int(i/N*255) # red
        view[i, j, 1] = 158          # green
        view[i, j, 2] = int(j/N*255) # blue
        view[i, j, 3] = 255          # alpha
        
p = figure(x_range=[0,10], y_range=[0,10])

p.image_rgba(image=[img], x=[0], y=[0], dw=[10], dh=[10])

show(p)

CustomJS for Widgets¶

In [13]:
from bokeh.layouts import row
from bokeh.models import CustomJS, ColumnDataSource, Slider

x = [i*0.005 for i in range(0, 200)]
y = x

source = ColumnDataSource(data=dict(x=x, y=y))

plot = figure(plot_width=400, plot_height=400)
plot.line('x', 'y', source=source, line_width=3, line_alpha=0.6)

# 支持es6语法
callback = CustomJS(args=dict(source=source), code="""
        var data = source.data;
        var f = cb_obj.value
        x = data['x']
        y = data['y']
        for (let i in x) {
            y[i] = Math.pow(x[i], f)
        }
        source.change.emit();
    """)

slider = Slider(start=0.1, end=4, value=1, step=.1, title="power", callback=callback)

layout = row(plot, slider)
show(layout)

import numpy as np
from bokeh.layouts import row, widgetbox
from bokeh.models import CustomJS, Slider
from bokeh.plotting import figure, show, ColumnDataSource

x = np.linspace(0, 10, 500)
y = 5*np.sin(3*x)

source = ColumnDataSource(data=dict(x=x, y=y))

plot = figure(y_range=(-10, 10), plot_width=400, plot_height=400)

plot.line('x', 'y', source=source, line_width=3, line_alpha=0.6)

cb = CustomJS(args=dict(source=source), code="""
    var data = source.data;
    var A = amp.value;
    var k = freq.value;
    var phi = phase.value;
    var B = offset.value;
    x = data['x']
    y = data['y']
    for (let i in x) {
        y[i] = B + A*Math.sin(k*x[i]+phi);
    }
    source.change.emit();
""")

amp_slider = Slider(start=0, end=10, value=5, step=.1,
                    title="Amplitude", callback=cb)
cb.args["amp"] = amp_slider

freq_slider = Slider(start=0, end=10, value=3, step=.1,
                     title="Frequency", callback=cb)
cb.args["freq"] = freq_slider

phase_slider = Slider(start=0, end=6.4, value=0, step=.1,
                      title="Phase", callback=cb)
cb.args["phase"] = phase_slider

offset_slider = Slider(start=-5, end=5, value=0, step=.1,
                       title="Offset", callback=cb)
cb.args["offset"] = offset_slider

layout = row(
    plot,
    widgetbox(amp_slider, freq_slider, phase_slider, offset_slider),
)

show(layout)
